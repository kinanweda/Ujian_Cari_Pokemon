show databases;

use world;

show tables;

select * from city;
select * from country;
select * from countrylanguage;

No.1
select * from city where (select CountryCode = 'IDN' where CountryCode = 'IDN') order by Population desc limit 10;

No.2
select c.ID as id, c.Name as nama_kota, c.District as district, t.Name as negara, c.population 
from city c inner join country t 
on c.CountryCode = t.Code order by Population desc limit 10;

No.3
select Code, Name, Continent, Region,IndepYear as tahun_merdeka from country where (select IndepYear where IndepYear is not null) order by tahun_merdeka asc limit 10;

No.4
select Continent as Benua, count(name) as Jumlah_Negara, sum(Population) as Populasi, avg(LifeExpectancy) as Rata2_kehidupan from country group by Continent having count(Continent) > 10 order by Populasi desc;

No.5
select Name as Nama, Continent as Benua, LifeExpectancy as AngkaHarapanHidup, GNP from country where continent = 'Asia' order by LifeExpectancy desc limit 9;

No.6
select l.CountryCode as countrycode, c.Name as name, Language, isOfficial, Percentage as percentage from countrylanguage l inner join country c on l.CountryCode = c.Code where Language = 'English' and isOfficial = 'T' order by percentage desc limit 10;

No.7
select t.Name as Negara_ASEAN, t.Population as Populasi_Negara, t.GNP, c.Name as Ibukota, c.Population as Populasi_Ibukota 
from city c inner join  country t
on c.ID = t.Capital
where t.Name in ('Brunei','Cambodia','East Timor','Indonesia','Laos','Malaysia','Myanmar','Philippines','Singapore','Thailand','Vietnam') 
order by Negara_ASEAN asc;

No.8
select t.Name as Negara_G20, t.Population as Populasi_Negara, t.GNP, c.Name as Ibukota, c.Population as Populasi_Ibukota 
from city c inner join  country t
on c.ID = t.Capital
where t.Name in ('Argentina','Australia','Brazil','Canada','China','France','Germany','India','Indonesia','Italia','Japan','Mexico','Russian Federation','Saudi Arabia','South Africa','South Korea','Turkey','United Kingdom','United States') 
order by Negara_G20 asc;
